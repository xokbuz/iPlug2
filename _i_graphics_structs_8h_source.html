310  void GetHSLA(float& h, float& s, float& l, float& a) const
311  {
312  const float fR = R / 255.f;
313  const float fG = G / 255.f;
314  const float fB = B / 255.f;
315  a = A / 255.f;
316 
317  const float fMin = std::min(fR, std::min(fG, fB));
318  const float fMax = std::max(fR, std::max(fG, fB));
319  const float fDiff = fMax - fMin;
320  const float fSum = fMax + fMin;
321 
322  l = 50.f * fSum;
323 
324  if (fMin == fMax) { s = 0.f; h = 0.f; l /= 100.f; return; }
325  else if (l < 50.f) { s = 100.f * fDiff / fSum; }
326  else { s = 100.f * fDiff / (2.f - fDiff); }
327 
328  if (fMax == fR) { h = 60.f * (fG - fB) / fDiff; }
329  if (fMax == fG) { h = 60.f * (fB - fR) / fDiff + 120.f; }
330  if (fMax == fB) { h = 60.f * (fR - fG) / fDiff + 240.f; }
331 
332  if (h < 0.f) { h = h + 360.f; }
333 
334  h /= 360.f;
335  s /= 100.f;
336  l /= 100.f;
337  }

431  static IColor FromHSLA(float h, float s, float l, float a = 1.f)
432  {
433  auto hue = [](float h, float m1, float m2) {
434  if (h < 0) h += 1;
435  if (h > 1) h -= 1;
436  if (h < 1.0f / 6.0f)
437  return m1 + (m2 - m1) * h * 6.0f;
438  else if (h < 3.0f / 6.0f)
439  return m2;
440  else if (h < 4.0f / 6.0f)
441  return m1 + (m2 - m1) * (2.0f / 3.0f - h) * 6.0f;
442  return m1;
443  };
444 
445  IColor col;
446  h = std::fmodf(h, 1.0f);
447  if (h < 0.0f) h += 1.0f;
448  s = Clip(s, 0.0f, 1.0f);
449  l = Clip(l, 0.0f, 1.0f);
450  float m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
451  float m1 = 2 * l - m2;
452  col.R = static_cast<int>(Clip(hue(h + 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
453  col.G = static_cast<int>(Clip(hue(h, m1, m2), 0.0f, 1.0f) * 255.f);
454  col.B = static_cast<int>(Clip(hue(h - 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
455  col.A = static_cast<int>(a * 255.f);
456  return col;
457  }
