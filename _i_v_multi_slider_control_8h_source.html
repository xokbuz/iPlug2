91  void SnapToMouse(float x, float y, EDirection direction, const IRECT& bounds, int valIdx = -1 /* TODO:: not used*/, double minClip = 0., double maxClip = 1.) override
92  {
93  bounds.Constrain(x, y);
94  int nVals = NVals();
95 
96  float value = 0.;
97  int sliderTest = -1;
98 
99  if(direction == EDirection::Vertical)
100  {
101  value = 1.f - (y-bounds.T) / bounds.H();
102 
103  for(auto i = 0; i < nVals; i++)
104  {
105  if(mTrackBounds.Get()[i].Contains(x, mTrackBounds.Get()[i].MH()))
106  {
107  sliderTest = i;
108  break;
109  }
110  }
111  }
112  else
113  {
114  value = (x-bounds.L) / bounds.W();
115 
116  for(auto i = 0; i < nVals; i++)
117  {
118  if(mTrackBounds.Get()[i].Contains(mTrackBounds.Get()[i].MW(), y))
119  {
120  sliderTest = i;
121  break;
122  }
123  }
124  }
125 
126  value = std::round( value / mGrain ) * mGrain;
127 
128  if (sliderTest > -1)
129  {
130  SetValue(mMinTrackValue + Clip(value, 0.f, 1.f) * (mMaxTrackValue - mMinTrackValue), sliderTest);
131  OnNewValue(sliderTest, static_cast<float>(GetValue(sliderTest)));
132 
133  mSliderHit = sliderTest;
134 
135  if (mPrevSliderHit != -1)
136  {
137  if (abs(mPrevSliderHit - mSliderHit) > 1 /*|| shiftClicked*/)
138  {
139  int lowBounds, highBounds;
140 
141  if (mPrevSliderHit < mSliderHit)
142  {
143  lowBounds = mPrevSliderHit;
144  highBounds = mSliderHit;
145  }
146  else
147  {
148  lowBounds = mSliderHit;
149  highBounds = mPrevSliderHit;
150  }
151 
152  for (auto i = lowBounds; i < highBounds; i++)
153  {
154  float frac = (float)(i - lowBounds) / float(highBounds-lowBounds);
155  SetValue(linearInterp(GetValue(lowBounds), GetValue(highBounds), frac), i);
156  OnNewValue(i, static_cast<float>(GetValue(i)));
157  }
158  }
159  }
160  mPrevSliderHit = mSliderHit;
161  }
162  else
163  {
164  mSliderHit = -1;
165  }
166
