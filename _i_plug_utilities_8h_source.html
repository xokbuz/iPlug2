291 static void MidiNoteName(double midiPitch, WDL_String& noteName, bool cents = false, bool middleCisC4 = false)
292 {
293  static const char noteNames[12][3] = {"C ","C#","D ","D#","E ","F ","F#","G ","G#","A ","A#","B "};
294 
295  int midiPitchR = (int) std::round(midiPitch);
296  int pitchClass = midiPitchR % 12;
297  int octave = (midiPitchR / 12) - (middleCisC4? 1 : 2);
298 
299  if (cents)
300  {
301  double frac = (midiPitch - (float) midiPitchR) * 100.;
302  noteName.SetFormatted(32, "%s%i %.0f", noteNames[pitchClass], octave, frac);
303  }
304  else
305  {
306  noteName.SetFormatted(32, "%s%i", noteNames[pitchClass], octave);
307  }
308 }
