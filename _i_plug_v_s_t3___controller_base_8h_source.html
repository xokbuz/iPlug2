33  void Initialize(IPlugAPIBase* pPlug, Steinberg::Vst::ParameterContainer& parameters, bool plugIsInstrument, bool midiIn)
34  {
35  Steinberg::Vst::EditControllerEx1* pEditController = dynamic_cast<Steinberg::Vst::EditControllerEx1*>(pPlug);
36 
37  Steinberg::Vst::UnitInfo unitInfo;
38  unitInfo.id = Steinberg::Vst::kRootUnitId;
39  unitInfo.parentUnitId = Steinberg::Vst::kNoParentUnitId;
40  Steinberg::UString unitNameSetter(unitInfo.name, 128);
41  unitNameSetter.fromAscii("Root");
42 
43  Steinberg::Vst::UnitID unitID = Steinberg::Vst::kRootUnitId;
44 
45  #ifdef VST3_PRESET_LIST
46  if (pPlug->NPresets())
47  {
48  unitInfo.programListId = kPresetParam;
49  parameters.addParameter(new IPlugVST3PresetParameter(pPlug->NPresets()));
50 
51  Steinberg::Vst::ProgramListWithPitchNames* pList = new Steinberg::Vst::ProgramListWithPitchNames(STR16("Factory Presets"), 0 /* list id */, Steinberg::Vst::kRootUnitId);
52 
53  Steinberg::Vst::String128 programName;
54  Steinberg::Vst::String128 pitchName;
55 
56  for (int programIdx=0; programIdx<pPlug->NPresets(); programIdx++)
57  {
58  Steinberg::UString(programName, str16BufferSize(Steinberg::Vst::String128)).assign(pPlug->GetPresetName(programIdx));
59  pList->addProgram (programName);
60 
61  //Set named notes. This could be different per-preset in VST3
62  for (int pitch = 0; pitch < 128; pitch++)
63  {
64  char pNoteText[32] = "";
65  if(pPlug->GetMidiNoteText(pitch, pNoteText))
66  {
67  Steinberg::UString(pitchName, str16BufferSize(Steinberg::Vst::String128)).assign(pNoteText);
68  pList->setPitchName(programIdx, pitch, pitchName);
69  }
70  }
71  }
72 
73  pEditController->addProgramList(pList);
74  }
75  else
76  #endif
77  unitInfo.programListId = Steinberg::Vst::kNoProgramListId;
78 
79  if (!plugIsInstrument)
80  parameters.addParameter(mBypassParameter = new IPlugVST3BypassParameter());
81 
82  pEditController->addUnit(new Steinberg::Vst::Unit(unitInfo));
83 
84  for (int i = 0; i < pPlug->NParams(); i++)
85  {
86  IParam* pParam = pPlug->GetParam(i);
87  unitID = Steinberg::Vst::kRootUnitId; // reset unitID
88 
89  const char* paramGroupName = pParam->GetGroupForHost();
90 
91  if (CStringHasContents(paramGroupName)) // if the parameter has a group
92  {
93  for (int j = 0; j < pPlug->NParamGroups(); j++) // loop through previously added groups
94  {
95  if (strcmp(paramGroupName, pPlug->GetParamGroupName(j)) == 0) // if group name found in existing groups
96  {
97  unitID = j + 1; // increment unitID
98  }
99  }
100 
101  if (unitID == Steinberg::Vst::kRootUnitId) // if unitID was still kRootUnitId, we found a new group, so add it and add the unit
102  {
103  unitID = pPlug->AddParamGroup(paramGroupName); // updates unitID
104  unitInfo.id = unitID;
105  unitInfo.parentUnitId = Steinberg::Vst::kRootUnitId;
106  unitInfo.programListId = Steinberg::Vst::kNoProgramListId;
107  unitNameSetter.fromAscii(paramGroupName);
108  pEditController->addUnit (new Steinberg::Vst::Unit (unitInfo));
109  }
110  }
111 
112  Steinberg::Vst::Parameter* pVST3Parameter = new IPlugVST3Parameter(pParam, i, unitID);
113  parameters.addParameter(pVST3Parameter);
114  }
115 
116  assert(VST3_NUM_CC_CHANS <= VST3_NUM_MIDI_IN_CHANS && "VST3_NUM_CC_CHANS must be less than or equal to VST3_NUM_MIDI_IN_CHANS");
117 
118 #if VST3_NUM_CC_CHANS > 0
119  if (midiIn)
120  {
121  unitInfo.id = unitID = pEditController->getUnitCount() + 1;
122  unitInfo.parentUnitId = Steinberg::Vst::kRootUnitId;
123  unitInfo.programListId = Steinberg::Vst::kNoProgramListId;
124  unitNameSetter.fromAscii(VST3_CC_UNITNAME);
125  pEditController->addUnit(new Steinberg::Vst::Unit(unitInfo));
126 
127  Steinberg::Vst::ParamID paramIdx = kMIDICCParamStartIdx;
128 
129  WDL_String chanGroupStr;
130  Steinberg::Vst::UnitID midiCCsUnitID = unitID;
131 
132  for (int chan = 0; chan < VST3_NUM_CC_CHANS; chan++)
133  {
134  chanGroupStr.SetFormatted(32, "CH%i", chan + 1);
135 
136  unitInfo.id = unitID = pEditController->getUnitCount() + 1;
137  unitInfo.parentUnitId = midiCCsUnitID;
138  unitInfo.programListId = Steinberg::Vst::kNoProgramListId;
139  unitNameSetter.fromAscii(chanGroupStr.Get());
140  pEditController->addUnit(new Steinberg::Vst::Unit(unitInfo));
141  // add 128 MIDI CCs
142  Steinberg::Vst::String128 paramName;
143  for (int i = 0; i < 128; i++)
144  {
145  Steinberg::UString(paramName, str16BufferSize(Steinberg::Vst::String128)).assign(IMidiMsg::CCNameStr(i));
146  parameters.addParameter(paramName, STR16(""), 0, 0, 0, paramIdx++, unitID);
147  }
148 
149  parameters.addParameter(STR16("Channel Aftertouch"), STR16(""), 0, 0, 0, paramIdx++, unitID);
150  parameters.addParameter(STR16("Pitch Bend"), STR16(""), 0, 0.5, 0, paramIdx++, unitID);
151  }
152  }
153 #endif
154  }

168  void PLUGIN_API SetParamNormalized(IPlugAPIBase* pPlug, Steinberg::Vst::ParamID tag, Steinberg::Vst::ParamValue value)
169  {
170  if (tag >= kBypassParam)
171  {
172  switch (tag)
173  {
174  case kPresetParam:
175  {
176  pPlug->RestorePreset(std::round((pPlug->NPresets() - 1) * value));
177  break;
178  }
179  default:
180  break;
181  }
182  }
183  else
184  {
185  IParam* pParam = pPlug->GetParam(tag);
186 
187  if (pParam)
188  {
189  pParam->SetNormalized(value);
190  pPlug->OnParamChangeUI(tag, kHost);
191 
192 // in VST3, parameter changes are managed by the host
193 #if !defined VST3C_API // && !defined VST3_API //TODO
194  pPlug->SendParameterValueFromAPI(tag, value, true);
195 #else
196  pPlug->SendParameterValueFromDelegate(tag, value, true);
197 #endif
198  }
199  }
200  }
