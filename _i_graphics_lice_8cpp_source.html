81 LICE_pixel LiceColor(const IColor& color)
82 {
83  auto preMul = [](int color, int A) {return (color * (A + 1)) >> 8; };
84  return LICE_RGBA(preMul(color.R, color.A), preMul(color.G, color.A), preMul(color.B, color.A), color.A);

94 {
95  if (!pBlend)

115 // Utilities for pre-multiplied blits (LICE assumes sources are not pre-multiplied)
116 static inline void PreMulCompositeSourceOver(LICE_pixel_chan* out, LICE_pixel_chan* in)
117 {
118  unsigned int alphaCmp = 256 - in[LICE_PIXEL_A];
119 
120  unsigned int A = in[LICE_PIXEL_A] + ((out[LICE_PIXEL_A] * alphaCmp) >> 8);
121  unsigned int R = in[LICE_PIXEL_R] + ((out[LICE_PIXEL_R] * alphaCmp) >> 8);
122  unsigned int G = in[LICE_PIXEL_G] + ((out[LICE_PIXEL_G] * alphaCmp) >> 8);
123  unsigned int B = in[LICE_PIXEL_B] + ((out[LICE_PIXEL_B] * alphaCmp) >> 8);
124 
125  _LICE_MakePixelClamp(out, R, G, B, A);
126 }
127 
128 static inline void PreMulCompositeAdd(LICE_pixel_chan* out, LICE_pixel_chan* in)
129 {
130  unsigned int alpha = in[LICE_PIXEL_A];
131 
132  unsigned int A = out[LICE_PIXEL_A] + (255 * in[LICE_PIXEL_A] / alpha);
133  unsigned int R = out[LICE_PIXEL_R] + (255 * in[LICE_PIXEL_R] / alpha);
134  unsigned int G = out[LICE_PIXEL_G] + (255 * in[LICE_PIXEL_G] / alpha);
135  unsigned int B = out[LICE_PIXEL_B] + (255 * in[LICE_PIXEL_B] / alpha);
136 
137  _LICE_MakePixelClamp(out, R, G, B, A);
138 }

140 static void PreMulBlit(LICE_IBitmap *dest, LICE_IBitmap *src, int dstx, int dsty, int srcx, int srcy, int srcw, int srch, float alpha, int mode)
141 {
142  srcx = dstx < 0 ? srcx - dstx : srcx;
143  srcy = dsty < 0 ? srcy - dsty : srcy;
144  dstx = std::max(dstx, 0);
145  dsty = std::max(dsty, 0);
146  srcw = std::min(srcw, dest->getWidth() - dstx);
147  srch = std::min(srch, dest->getHeight() - dsty);
148 
149  int inStride = src->getRowSpan() * 4;
150  int outStride = dest->getRowSpan() * 4;
151 
152  LICE_pixel_chan* in = ((LICE_pixel_chan*) src->getBits()) + (srcy * inStride) + (srcx * 4);
153  LICE_pixel_chan* out = ((LICE_pixel_chan*) dest->getBits()) + (dsty * outStride) + (dstx * 4);
154 
155  if ((mode & LICE_BLIT_MODE_MASK) == LICE_BLIT_MODE_ADD)
156  {
157  for (int i = 0; i < srch; i++, in += inStride, out += outStride)
158  {
159  for (int j = 0; j < srcw; j++)
160  PreMulCompositeAdd(out + j * 4, in + j * 4);
161  }
162  }
163  else
164  {
165  for (int i = 0; i < srch; i++, in += inStride, out += outStride)
166  {
167  for (int j = 0; j < srcw; j++)
168  PreMulCompositeSourceOver(out + j * 4, in + j * 4);
169  }
170  }
171 }

242 void IGraphicsLice::DrawBitmap(const IBitmap& bitmap, const IRECT& bounds, int srcX, int srcY, const IBlend* pBlend)
243 {
244  bool preMultiplied = static_cast<Bitmap*>(bitmap.GetAPIBitmap())->IsPreMultiplied();
245  const int ds = GetScreenScale();
246 
247  IRECT sr = TransformRECT(bounds);
248  IRECT r = sr.Intersect(mDrawRECT.GetScaled(ds));
249 
250  srcX = (srcX * ds) + r.L - sr.L;
251  srcY = (srcY * ds) + r.T - sr.T;
252 
253  if (preMultiplied)
254  PreMulBlit(mRenderBitmap, bitmap.GetAPIBitmap()->GetBitmap(), r.L, r.T, srcX, srcY, r.W(), r.H(), BlendWeight(pBlend), LiceBlendMode(pBlend));
255  else
256  LICE_Blit(mRenderBitmap, bitmap.GetAPIBitmap()->GetBitmap(), r.L, r.T, srcX, srcY, r.W(), r.H(), BlendWeight(pBlend), LiceBlendMode(pBlend));

483 void IGraphicsLice::FillArc(const IColor& color, float cx, float cy, float r, float a1, float a2, const IBlend* pBlend)
484 {
485  NeedsClipping();
486 
487  if (a2 < a1)
488  std::swap(a1, a2);
489 
490  if (a2 >= a1 + 360.f)
491  {
492  FillCircle(color, cx, cy, r, pBlend);
493  return;
494  }
495 
496  float xarray[181];
497  float yarray[181];
498 
499  if (a2 > a1 + 180.f)
500  {
501  if (!OpacityCheck(color, pBlend))
502  {
503  OpacityLayer(&IGraphicsLice::FillArc, pBlend, color, cx, cy, r, a1, a2, nullptr);
504  return;
505  }
506 
507  FillArc(color, cx, cy, r, a1 + 178.f, a2, pBlend);
508  a2 = a1 + 180.f;
509  }
510 
511  a1 = DegToRad(a1-90.f);
512  a2 = DegToRad(a2-90.f);
513 
514  int arcpoints = 180.0 * std::min(1., (a2 - a1) / PI);
515  double arcincrement = (a2 - a1) / arcpoints;
516  for(int i = 0; i < arcpoints; i++)
517  {
518  xarray[i] = cx + cosf(i * arcincrement + a1) * r;
519  yarray[i] = cy + sinf(i * arcincrement + a1) * r;
520  }
521 
522  xarray[arcpoints] = cx;
523  yarray[arcpoints] = cy;
524 
525  FillConvexPolygon(color, xarray, yarray, arcpoints + 1, pBlend);
526 }

854 void IGraphicsLice::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
855 {
856  const APIBitmap* pBitmap = layer->GetAPIBitmap();
857  LICE_IBitmap* pLayerBitmap = pBitmap->GetBitmap();
858 
859  int stride = pLayerBitmap->getRowSpan() * 4;
860  int size = pLayerBitmap->getHeight() * stride;
861 
862  if (mask.GetSize() >= size)
863  {
864  int x = std::round(shadow.mXOffset * GetScreenScale());
865  int y = std::round(shadow.mYOffset * GetScreenScale());
866  int nRows = pBitmap->GetHeight() - std::abs(y);
867  int nCols = pBitmap->GetWidth() - std::abs(x);
868  LICE_pixel_chan* in = mask.Get() + (std::max(-x, 0) * 4) + (std::max(-y, 0) * stride);
869  LICE_pixel_chan* out = ((LICE_pixel_chan*) pLayerBitmap->getBits()) + (std::max(x, 0) * 4) + (std::max(y, 0) * stride);
870 
871  // Pre-multiply color components
872  IColor color = shadow.mPattern.GetStop(0).mColor;
873  color.Clamp();
874  unsigned int ia = (color.A * static_cast<int>(Clip(shadow.mOpacity, 0.f, 1.f) * 255.0));
875  unsigned int ir = (color.R * ia);
876  unsigned int ig = (color.G * ia);
877  unsigned int ib = (color.B * ia);
878 
879  if (!shadow.mDrawForeground)
880  {
881  LICE_Clear(pLayerBitmap, 0);
882 
883  for (int i = 0; i < nRows; i++, in += stride, out += stride)
884  {
885  LICE_pixel_chan* chans = out;
886 
887  for (int j = 0; j < nCols; j++, chans += 4)
888  {
889  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
890 
891  unsigned int A = (ia * maskAlpha) >> 16;
892  unsigned int R = (ir * maskAlpha) >> 16;
893  unsigned int G = (ig * maskAlpha) >> 16;
894  unsigned int B = (ib * maskAlpha) >> 16;
895 
896  _LICE_MakePixelNoClamp(chans, R, G, B, A);
897  }
898  }
899  }
900  else
901  {
902  for (int i = 0; i < nRows; i++, in += stride, out += stride)
903  {
904  LICE_pixel_chan* chans = out;
905 
906  for (int j = 0; j < nCols; j++, chans += 4)
907  {
908  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
909  unsigned int alphaCmp = 255 - chans[LICE_PIXEL_A];
910 
911  unsigned int A = chans[LICE_PIXEL_A] + ((alphaCmp * ia * maskAlpha) >> 24);
912  unsigned int R = chans[LICE_PIXEL_R] + ((alphaCmp * ir * maskAlpha) >> 24);
913  unsigned int G = chans[LICE_PIXEL_G] + ((alphaCmp * ig * maskAlpha) >> 24);
914  unsigned int B = chans[LICE_PIXEL_B] + ((alphaCmp * ib * maskAlpha) >> 24);
915 
916  _LICE_MakePixelClamp(chans, R, G, B, A);
917  }
918  }
919  }
920  }
921 }
