440  IPattern GetSVGPattern(const NSVGpaint& paint, float opacity)
441  {
442  int alpha = std::min(255, std::max(int(roundf(opacity * 255.f)), 0));
443 
444  switch (paint.type)
445  {
446  case NSVG_PAINT_COLOR:
447  return IColor(alpha, (paint.color >> 0) & 0xFF, (paint.color >> 8) & 0xFF, (paint.color >> 16) & 0xFF);
448 
449  case NSVG_PAINT_LINEAR_GRADIENT:
450  case NSVG_PAINT_RADIAL_GRADIENT:
451  {
452  NSVGgradient* pGrad = paint.gradient;
453 
454  IPattern pattern(paint.type == NSVG_PAINT_LINEAR_GRADIENT ? EPatternType::Linear : EPatternType::Radial);
455 
456  // Set Extend Rule
457  switch (pGrad->spread)
458  {
459  case NSVG_SPREAD_PAD: pattern.mExtend = EPatternExtend::Pad; break;
460  case NSVG_SPREAD_REFLECT: pattern.mExtend = EPatternExtend::Reflect; break;
461  case NSVG_SPREAD_REPEAT: pattern.mExtend = EPatternExtend::Repeat; break;
462  }
463 
464  // Copy Stops
465  for (int i = 0; i < pGrad->nstops; i++)
466  {
467  unsigned int color = pGrad->stops[i].color;
468  pattern.AddStop(IColor(255, (color >> 0) & 0xFF, (color >> 8) & 0xFF, (color >> 16) & 0xFF), pGrad->stops[i].offset);
469  }
470 
471  // Copy transform
472  pattern.SetTransform(pGrad->xform[0], pGrad->xform[1], pGrad->xform[2], pGrad->xform[3], pGrad->xform[4], pGrad->xform[5]);
473 
474  return pattern;
475  }
476  default:
477  return IColor(alpha, 0, 0, 0);
478  }

481  void DoDrawSVG(const ISVG& svg, const IBlend* pBlend = nullptr)
482  {
483 #ifdef IGRAPHICS_SKIA
484  SkCanvas* canvas = static_cast<SkCanvas*>(GetDrawContext());
485  svg.mSVGDom->render(canvas); //TODO: blend
486 #else
487  NSVGimage* pImage = svg.mImage;
488 
489  assert(pImage != nullptr);
490 
491  for (NSVGshape* pShape = pImage->shapes; pShape; pShape = pShape->next)
492  {
493  if (!(pShape->flags & NSVG_FLAGS_VISIBLE))
494  continue;
495 
496  // Build a new path for each shape
497  PathClear();
498 
499  // iterate subpaths in this shape
500  for (NSVGpath* pPath = pShape->paths; pPath; pPath = pPath->next)
501  {
502  PathMoveTo(pPath->pts[0], pPath->pts[1]);
503 
504  for (int i = 1; i < pPath->npts; i += 3)
505  {
506  float *p = &pPath->pts[i*2];
507  PathCubicBezierTo(p[0], p[1], p[2], p[3], p[4], p[5]);
508  }
509 
510  if (pPath->closed)
511  PathClose();
512 
513  // Compute whether this path is a hole or a solid and set the winding direction accordingly.
514  int crossings = 0;
515  IVec2 p0{pPath->pts[0], pPath->pts[1]};
516  IVec2 p1{pPath->bounds[0] - 1.0f, pPath->bounds[1] - 1.0f};
517  // Iterate all other paths
518  for (NSVGpath *pPath2 = pShape->paths; pPath2; pPath2 = pPath2->next)
519  {
520  if (pPath2 == pPath)
521  continue;
522  // Iterate all lines on the path
523  if (pPath2->npts < 4)
524  continue;
525  for (int i = 1; i < pPath2->npts + 3; i += 3)
526  {
527  float *p = &pPath2->pts[2*i];
528  // The previous point
529  IVec2 p2 {p[-2], p[-1]};
530  // The current point
531  IVec2 p3 = (i < pPath2->npts) ? IVec2{p[4], p[5]} : IVec2{pPath2->pts[0], pPath2->pts[1]};
532  float crossing = GetLineCrossing(p0, p1, p2, p3);
533  float crossing2 = GetLineCrossing(p2, p3, p0, p1);
534  if (0.0 <= crossing && crossing < 1.0 && 0.0 <= crossing2)
535  {
536  crossings++;
537  }
538  }
539  }
540  PathSetWinding(crossings % 2 != 0);
541  }
542 
543  // Fill combined path using windings set in subpaths
544  if (pShape->fill.type != NSVG_PAINT_NONE)
545  {
546  IFillOptions options;
547  options.mFillRule = EFillRule::Preserve;
548 
549  options.mPreserve = pShape->stroke.type != NSVG_PAINT_NONE;
550  PathFill(GetSVGPattern(pShape->fill, pShape->opacity), options, pBlend);
551  }
552 
553  // Stroke
554  if (pShape->stroke.type != NSVG_PAINT_NONE)
555  {
556  IStrokeOptions options;
557 
558  options.mMiterLimit = pShape->miterLimit;
559 
560  switch (pShape->strokeLineCap)
561  {
562  case NSVG_CAP_BUTT: options.mCapOption = ELineCap::Butt; break;
563  case NSVG_CAP_ROUND: options.mCapOption = ELineCap::Round; break;
564  case NSVG_CAP_SQUARE: options.mCapOption = ELineCap::Square; break;
565  }
566 
567  switch (pShape->strokeLineJoin)
568  {
569  case NSVG_JOIN_MITER: options.mJoinOption = ELineJoin::Miter; break;
570  case NSVG_JOIN_ROUND: options.mJoinOption = ELineJoin::Round; break;
571  case NSVG_JOIN_BEVEL: options.mJoinOption = ELineJoin::Bevel; break;
572  }
573 
574  options.mDash.SetDash(pShape->strokeDashArray, pShape->strokeDashOffset, pShape->strokeDashCount);
575 
576  PathStroke(GetSVGPattern(pShape->stroke, pShape->opacity), pShape->strokeWidth, options, pBlend);
577  }
578  }
579  #endif
