196 void IControl::SetDirty(bool triggerAction, int valIdx)
197 {
198  valIdx = (NVals() == 1) ? 0 : valIdx;
199 
200  auto setValue = [this](int v) { SetValue(Clip(GetValue(v), 0.0, 1.0), v); };
201  ForValIdx(valIdx, setValue);
202 
203  mDirty = true;
204 
205  if (triggerAction)
206  {
207  auto paramUpdate = [this](int v)
208  {
209  if (GetParamIdx(v) > kNoParameter)
210  {
211  GetDelegate()->SendParameterValueFromUI(GetParamIdx(v), GetValue(v)); //TODO: take tuple
212  GetUI()->UpdatePeers(this, v);
213  }
214  };
215 
216  ForValIdx(valIdx, paramUpdate);
217 
218  if (mActionFunc)
219  mActionFunc(this);
220  }
221 }

339 void IControl::PromptUserInput(int valIdx)
340 {
341  if (valIdx > kNoValIdx && GetParamIdx(valIdx) > kNoParameter && !mDisablePrompt)
342  {
343  if (GetParam(valIdx)->NDisplayTexts()) // popup menu
344  {
345  GetUI()->PromptUserInput(*this, mRECT, valIdx);
346  }
347  else // text entry
348  {
349  float cX = mRECT.MW();
350  float cY = mRECT.MH();
351  float halfW = PARAM_EDIT_W/2.f;
352  float halfH = PARAM_EDIT_H/2.f;
353 
354  IRECT txtRECT = IRECT(cX - halfW, cY - halfH, cX + halfW,cY + halfH);
355  GetUI()->PromptUserInput(*this, txtRECT, valIdx);
356  }
357 
358  SetDirty(false);
359  }
360 }

402 void IControl::SnapToMouse(float x, float y, EDirection direction, const IRECT& bounds, int valIdx, double minClip, double maxClip)
403 {
404  bounds.Constrain(x, y);
405 
406  float val;
407 
408  if(direction == EDirection::Vertical)
409  val = 1.f - (y-bounds.T) / bounds.H();
410  else
411  val = (x-bounds.L) / bounds.W();
412 
413  auto valFunc = [&](int valIdx) {
414  SetValue(Clip(std::round(val / 0.001 ) * 0.001, minClip, maxClip), valIdx);
415  };
416 
417  ForValIdx(valIdx, valFunc);
418  SetDirty(true, valIdx);
419 }

848 void IDirBrowseControlBase::ScanDirectory(const char* path, IPopupMenu& menuToAddTo)
849 {
850  WDL_DirScan d;
851 
852  if (!d.First(path))
853  {
854  do
855  {
856  const char* f = d.GetCurrentFN();
857  if (f && f[0] != '.')
858  {
859  if (d.GetCurrentIsDirectory())
860  {
861  WDL_String subdir;
862  d.GetCurrentFullFN(&subdir);
863  IPopupMenu* pNewMenu = new IPopupMenu();
864  menuToAddTo.AddItem(d.GetCurrentFN(), pNewMenu, -2);
865  ScanDirectory(subdir.Get(), *pNewMenu);
866  }
867  else
868  {
869  const char* a = strstr(f, mExtension.Get());
870  if (a && a > f && strlen(a) == strlen(mExtension.Get()))
871  {
872  WDL_String menuEntry {f};
873 
874  if(!mShowFileExtensions)
875  menuEntry.Set(f, (int) (a - f));
876 
877  IPopupMenu::Item* pItem = new IPopupMenu::Item(menuEntry.Get(), IPopupMenu::Item::kNoFlags, mFiles.GetSize());
878  menuToAddTo.AddItem(pItem, -2 /* sort alphabetically */);
879  WDL_String* pFullPath = new WDL_String("");
880  d.GetCurrentFullFN(pFullPath);
881  mFiles.Add(pFullPath);
882  }
883  }
884  }
885  } while (!d.Next());
886  }
887 
888  if(!mShowEmptySubmenus)
889  menuToAddTo.RemoveEmptySubmenus();
890 }
