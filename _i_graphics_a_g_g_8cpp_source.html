87 agg::comp_op_e AGGBlendMode(const IBlend* pBlend)
88 {
89  if (!pBlend)
90  return agg::comp_op_src_over;
91 
92  switch (pBlend->mMethod)
93  {
94  case EBlend::SrcOver: return agg::comp_op_src_over;
95  case EBlend::SrcIn: return agg::comp_op_src_in;
96  case EBlend::SrcOut: return agg::comp_op_src_out;
97  case EBlend::SrcAtop: return agg::comp_op_src_atop;
98  case EBlend::DstOver: return agg::comp_op_dst_over;
99  case EBlend::DstIn: return agg::comp_op_dst_in;
100  case EBlend::DstOut: return agg::comp_op_dst_out;
101  case EBlend::DstAtop: return agg::comp_op_dst_atop;
102  case EBlend::Add: return agg::comp_op_plus;
103  case EBlend::XOR: return agg::comp_op_xor;
104  }

127 template <typename Rasterizer, typename FuncType, typename ColorArrayType>
128 static void GradientRasterize(Rasterizer& rasterizer, const FuncType& gradientFunc, agg::trans_affine& xform, ColorArrayType& colors, agg::comp_op_e op)
129 {
130  using InterpolatorType = agg::span_interpolator_linear<>;
131  using SpanGradientType = agg::span_gradient<agg::rgba8, InterpolatorType, FuncType, ColorArrayType>;
132 
133  InterpolatorType spanInterpolator(xform);
134  SpanGradientType spanGradient(spanInterpolator, gradientFunc, colors, 0, 512);
135  rasterizer.Rasterize(spanGradient, op);
136 }

138 template <typename Rasterizer, typename FuncType, typename ColorArrayType>
139 static void GradientRasterizeAdapt(Rasterizer& rasterizer, EPatternExtend extend, const FuncType& gradientFunc, agg::trans_affine& xform, ColorArrayType& colors, agg::comp_op_e op)
140 {
141  using reflect = agg::gradient_reflect_adaptor<FuncType>;
142  using repeat = agg::gradient_repeat_adaptor<FuncType>;
143 
144  switch (extend)
145  {
146  case EPatternExtend::None: //TODO: extend none
147  case EPatternExtend::Pad:
148  GradientRasterize(rasterizer, gradientFunc, xform, colors, op);
149  break;
150  case EPatternExtend::Reflect:
151  GradientRasterize(rasterizer, reflect(gradientFunc), xform, colors, op);
152  break;
153  case EPatternExtend::Repeat:
154  GradientRasterize(rasterizer, repeat(gradientFunc), xform, colors, op);
155  break;
156  }
157 }

159 void IGraphicsAGG::Rasterizer::Rasterize(const IPattern& pattern, agg::comp_op_e op, float opacity, EFillRule rule)
160 {
161  mRasterizer.filling_rule(rule == EFillRule::Winding ? agg::fill_non_zero : agg::fill_even_odd );
162 
163  switch (pattern.mType)
164  {
165  case EPatternType::Solid:
166  Rasterize(AGGColor(pattern.GetStop(0).mColor, opacity), op);
167  break;
168  case EPatternType::Linear:
169  case EPatternType::Radial:
170  {
171  // Common gradient objects
172  const IMatrix& m = pattern.mTransform;
173 
174  agg::trans_affine gradientMTX(m.mXX, m.mYX , m.mXY, m.mYY, m.mTX, m.mTY);
175  agg::gradient_lut<agg::color_interpolator<agg::rgba8>, 512> colors;
176 
177  // Scaling
178  gradientMTX = (agg::trans_affine() / mGraphics.mTransform) * gradientMTX * agg::trans_affine_scaling(512.0);
179 
180  // Make gradient lut
181  colors.remove_all();
182 
183  for (int i = 0; i < pattern.NStops(); i++)
184  {
185  const IColorStop& stop = pattern.GetStop(i);
186  float offset = stop.mOffset;
187  colors.add_color(offset, AGGColor(stop.mColor, opacity));
188  }
189 
190  colors.build_lut();
191 
192  // Rasterize
193  if (pattern.mType == EPatternType::Linear)
194  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_y(), gradientMTX, colors, op);
195  else
196  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_radial_d(), gradientMTX, colors, op);
197  }
198  break;
199  }
200 }

261 bool CheckTransform(const agg::trans_affine& mtx)
262 {
263  if (!agg::is_equal_eps(mtx.tx - std::round(mtx.tx), 0.0, 1e-3))
264  return false;
265  if (!agg::is_equal_eps(mtx.ty - std::round(mtx.ty), 0.0, 1e-3))
266  return false;
267 
268  agg::trans_affine mtx_without_translate(mtx);
269  mtx_without_translate.tx = mtx_without_translate.ty = 0.0;
270 
271  return mtx_without_translate.is_identity(1e-3);
272 }

274 void IGraphicsAGG::DrawBitmap(const IBitmap& bitmap, const IRECT& dest, int srcX, int srcY, const IBlend* pBlend)
275 {
276  bool preMultiplied = static_cast<Bitmap*>(bitmap.GetAPIBitmap())->IsPreMultiplied();
277  IRECT bounds = mClipRECT.Intersect(dest);
278  bounds.Scale(GetBackingPixelScale());
279 
280  APIBitmap* pAPIBitmap = dynamic_cast<Bitmap*>(bitmap.GetAPIBitmap());
281  agg::pixel_map* pSource = pAPIBitmap->GetBitmap();
282  agg::rendering_buffer src(pSource->buf(), pSource->width(), pSource->height(), pSource->row_bytes());
283 
284  agg::trans_affine srcMtx;
285  srcMtx /= mTransform;
286  srcMtx *= agg::trans_affine_translation(srcX - dest.L, srcY - dest.T);
287  srcMtx *= agg::trans_affine_scaling(bitmap.GetScale() * bitmap.GetDrawScale());
288 
289  if (bounds.IsPixelAligned() && CheckTransform(srcMtx))
290  {
291  double scale = GetScreenScale() * GetScreenScale() / pAPIBitmap->GetScale();
292  IRECT destScaled = dest.GetScaled(GetBackingPixelScale());
293  srcX = std::round(srcX * scale + std::max(0.f, bounds.L - destScaled.L));
294  srcY = std::round(srcY * scale + std::max(0.f, bounds.T - destScaled.T));
295  bounds.Translate(mTransform.tx, mTransform.ty);
296 
297  mRasterizer.BlendFrom(src, bounds, srcX, srcY, AGGBlendMode(pBlend), AGGCover(pBlend), preMultiplied);
298  }
299  else
300  {
301  agg::rounded_rect rect(dest.L, dest.T, dest.R, dest.B, 0);
302  agg::conv_transform<agg::rounded_rect> tr(rect, mTransform);
303 
304  if (preMultiplied)
305  {
306  PixfmtPreType fmtSrc(src);
307  mRasterizer.Rasterize(fmtSrc, tr, srcMtx, AGGBlendMode(pBlend), AGGCover(pBlend));
308  }
309  else
310  {
311  PixfmtType fmtSrc(src);
312  mRasterizer.Rasterize(fmtSrc, tr, srcMtx, AGGBlendMode(pBlend), AGGCover(pBlend));
313  }
314  }
315 }

399 void IGraphicsAGG::PathStroke(const IPattern& pattern, float thickness, const IStrokeOptions& options, const IBlend* pBlend)
400 {
401  using CPType = agg::conv_curve<agg::path_storage>;
402  using S1Type = agg::conv_transform<CPType>;
403  using S2Type = agg::conv_stroke<S1Type>;
404  using S3Type = agg::conv_transform<S2Type>;
405  using D2Type = agg::conv_dash<S1Type>;
406  using D3Type = agg::conv_stroke<D2Type>;
407  using D4Type = agg::conv_transform<D3Type>;
408 
409  agg::trans_affine tranform(mTransform);
410  CPType curvedPath(mPath);
411  S1Type basePath(curvedPath, tranform.invert());
412 
413  if (options.mDash.GetCount())
414  {
415  D2Type dashedPath(basePath);
416  D3Type strokedDashedPath(dashedPath);
417  D4Type finalPath(strokedDashedPath, mTransform);
418 
419  // Set the dashes (N.B. - for odd counts the array is read twice)
420  int dashCount = options.mDash.GetCount();
421  int dashMax = dashCount & 1 ? dashCount * 2 : dashCount;
422  const float* dashArray = options.mDash.GetArray();
423 
424  dashedPath.remove_all_dashes();
425  dashedPath.dash_start(options.mDash.GetOffset());
426 
427  for (int i = 0; i < dashMax; i += 2)
428  dashedPath.add_dash(dashArray[i % dashCount], dashArray[(i + 1) % dashCount]);
429 
430  StrokeOptions(strokedDashedPath, thickness, options);
431  mRasterizer.Rasterize(finalPath, pattern, AGGBlendMode(pBlend), BlendWeight(pBlend));
432  }
433  else
434  {
435  S2Type strokedPath(basePath);
436  S3Type finalPath(strokedPath, mTransform);
437 
438  StrokeOptions(strokedPath, thickness, options);
439  mRasterizer.Rasterize(finalPath, pattern, AGGBlendMode(pBlend), BlendWeight(pBlend));
440  }
441 
442  if (!options.mPreserve)
443  mPath.remove_all();
444 }

506 void IGraphicsAGG::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
507 {
508  const APIBitmap* pBitmap = layer->GetAPIBitmap();
509  agg::pixel_map* pPixMap = pBitmap->GetBitmap();
510  int size = pPixMap->height() * pPixMap->row_bytes();
511 
512  if (mask.GetSize() >= size)
513  {
514  if (!shadow.mDrawForeground)
515  {
516  pBitmap->GetBitmap()->clear(0);
517  }
518 
519  IRECT bounds(layer->Bounds());
520  agg::pixel_wrapper* shadowSource = new agg::pixel_wrapper(mask.Get(), pPixMap->width(), pPixMap->height(), pPixMap->bpp(), pPixMap->row_bytes());
521  APIBitmap* shadowBitmap = new Bitmap(shadowSource, pBitmap->GetScale(), pBitmap->GetDrawScale(), true);
522  IBitmap bitmap(shadowBitmap, 1, false);
523  ILayer shadowLayer(shadowBitmap, layer->Bounds(), nullptr, IRECT());
524 
525  PathTransformSave();
526  PushLayer(layer.get());
527  PushLayer(&shadowLayer);
528  PathRect(layer->Bounds());
529  IBlend blend1(EBlend::SrcIn, 1.0);
530  PathTransformTranslate(-shadow.mXOffset, -shadow.mYOffset);
531  PathFill(shadow.mPattern, IFillOptions(), &blend1);
532  PopLayer();
533  IBlend blend2(EBlend::DstOver, shadow.mOpacity);
534  bounds.Translate(shadow.mXOffset, shadow.mYOffset);
535  DrawBitmap(bitmap, bounds, 0, 0, &blend2);
536  PopLayer();
537  PathTransformRestore();
538  }

565 void IGraphicsAGG::PrepareAndMeasureText(const IText& text, const char* str, IRECT& r, double& x, double & y) const
566 {
567  StaticStorage<IFontData>::Accessor storage(sFontCache);
568  IFontData* pFont = storage.Find(text.mFont);
569 
570  if (!pFont || !SetFont(text.mFont, pFont))
571  {
572  assert(0 && "No font found - did you forget to load it?");
573  }
574 
575  const bool textHinting = false;
576 
577  // Set dpi to 72 to allow finer resolution of text sizes
578  mFontEngine.resolution(72);
579  mFontEngine.hinting(textHinting);
580  mFontEngine.height(text.mSize * pFont->GetHeightEMRatio());
581  mFontEngine.flip_y(true);
582 
583  const double textHeight = text.mSize;
584  const double EMHeight = pFont->GetAscender() - pFont->GetDescender();
585  const double ascender = text.mSize * pFont->GetAscender() / EMHeight;
586  const double descender = text.mSize * pFont->GetDescender() / EMHeight;
587 
588  mFontManager.reset_last_glyph();
589  double textWidth = 0.0;
590 
591  for (int i = 0; str[i]; i++)
592  {
593  const agg::glyph_cache* pGlyph = mFontManager.glyph(str[i]);
594 
595  if (textKerning)
596  {
597  double dx = 0.0;
598  double dy = 0.0;
599  mFontManager.add_kerning(&dx, &dy);
600  textWidth += dx;
601  }
602 
603  textWidth += pGlyph->advance_x;
604  }
605 
606  switch (text.mAlign)
607  {
608  case EAlign::Near: x = r.L; break;
609  case EAlign::Center: x = r.MW() - (textWidth / 2.0); break;
610  case EAlign::Far: x = r.R - textWidth; break;
611  }
612 
613  switch (text.mVAlign)
614  {
615  case EVAlign::Top: y = r.T + ascender; break;
616  case EVAlign::Middle: y = r.MH() + descender + (textHeight / 2.0); break;
