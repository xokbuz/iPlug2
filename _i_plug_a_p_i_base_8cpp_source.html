69 bool IPlugAPIBase::CompareState(const uint8_t* pIncomingState, int startPos) const
70 {
71  bool isEqual = true;
72 
73  const double* data = (const double*) pIncomingState + startPos;
74 
75  // dirty hack here because protools treats param values as 32 bit int and in IPlug they are 64bit float
76  // if we memcmp() the incoming state with the current they may have tiny differences due to the quantization
77  for (int i = 0; i < NParams(); i++)
78  {
79  float v = (float) GetParam(i)->Value();
80  float vi = (float) *(data++);
81 
82  isEqual &= (std::fabs(v - vi) < 0.00001);
83  }
84 
85  return isEqual;
86 }

147 void IPlugAPIBase::OnTimer(Timer& t)
148 {
149  if(HasUI())
150  {
151  // in distributed VST3, parameter changes are managed by the host
152  #if !defined VST3C_API && !defined VST3P_API // && !defined VST3_API
153  while(mParamChangeFromProcessor.ElementsAvailable())
154  {
155  ParamTuple p;
156  mParamChangeFromProcessor.Pop(p);
157  SendParameterValueFromDelegate(p.idx, p.value, false); // TODO: if the parameter hasn't changed maybe we shouldn't do anything?
158  }
159 
160  while (mMidiMsgsFromProcessor.ElementsAvailable())
161  {
162  IMidiMsg msg;
163  mMidiMsgsFromProcessor.Pop(msg);
164  SendMidiMsgFromDelegate(msg);
165  }
166 
167  while (mSysExDataFromProcessor.ElementsAvailable())
168  {
169  SysExData msg;
170  mSysExDataFromProcessor.Pop(msg);
171  SendSysexMsgFromDelegate({msg.mOffset, msg.mData, msg.mSize});
172  }
173  #endif
174 
175  // Midi messages from the processor to the controller, are sent as IMessages and SendMidiMsgFromDelegate gets triggered on the other side's notify
176  #if defined VST3P_API // || defined VST3_API
177  while (mMidiMsgsFromProcessor.ElementsAvailable())
178  {
179  IMidiMsg msg;
180  mMidiMsgsFromProcessor.Pop(msg);
181  TransmitMidiMsgFromProcessor(msg);
182  }
183 
184  while (mSysExDataFromProcessor.ElementsAvailable())
185  {
186  SysExData data;
187  mSysExDataFromProcessor.Pop(data);
188  TransmitSysExDataFromProcessor(data);
189  }
190  #endif
191  }
192 
193  OnIdle();
194 }
