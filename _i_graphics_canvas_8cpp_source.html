207 void IGraphicsCanvas::SetCanvasSourcePattern(val& context, const IPattern& pattern, const IBlend* pBlend)
208 {
209  SetCanvasBlendMode(context, pBlend);
210 
211  switch (pattern.mType)
212  {
213  case EPatternType::Solid:
214  {
215  const IColor color = pattern.GetStop(0).mColor;
216  std::string colorString = CanvasColor(color, BlendWeight(pBlend));
217 
218  context.set("fillStyle", colorString);
219  context.set("strokeStyle", colorString);
220  }
221  break;
222 
223  case EPatternType::Linear:
224  case EPatternType::Radial:
225  {
226  double x, y;
227  IMatrix m = IMatrix(pattern.mTransform).Invert();
228  m.TransformPoint(x, y, 0.0, 1.0);
229 
230  val gradient = (pattern.mType == EPatternType::Linear) ?
231  context.call<val>("createLinearGradient", m.mTX, m.mTY, x, y) :
232  context.call<val>("createRadialGradient", m.mTX, m.mTY, 0.0, m.mTX, m.mTY, m.mXX);
233 
234  for (int i = 0; i < pattern.NStops(); i++)
235  {
236  const IColorStop& stop = pattern.GetStop(i);
237  gradient.call<void>("addColorStop", stop.mOffset, CanvasColor(stop.mColor));
238  }
239 
240  context.set("fillStyle", gradient);
241  context.set("strokeStyle", gradient);
242  }
243  break;
244  }
245 }

267 void IGraphicsCanvas::PrepareAndMeasureText(const IText& text, const char* str, IRECT& r, double& x, double & y) const
268 {
269  StaticStorage<Font>::Accessor storage(sFontCache);
270  Font* pFont = storage.Find(text.mFont);
271 
272  assert(pFont && "No font found - did you forget to load it?");
273 
274  FontDescriptor descriptor = &pFont->mDescriptor;
275  val context = GetContext();
276  std::string fontString = GetFontString(descriptor->first.Get(), descriptor->second.Get(), text.mSize * pFont->mEMRatio);
277 
278  context.set("font", fontString);
279 
280  const double textWidth = context.call<val>("measureText", std::string(str))["width"].as<double>();
281  const double textHeight = text.mSize;
282  const double ascender = pFont->mAscenderRatio * textHeight;
283  const double descender = -(1.0 - pFont->mAscenderRatio) * textHeight;
284 
285  switch (text.mAlign)
286  {
287  case EAlign::Near: x = r.L; break;
288  case EAlign::Center: x = r.MW() - (textWidth / 2.0); break;
289  case EAlign::Far: x = r.R - textWidth; break;
290  }
291 
292  switch (text.mVAlign)
293  {
294  case EVAlign::Top: y = r.T + ascender; break;
295  case EVAlign::Middle: y = r.MH() + descender + (textHeight / 2.0); break;
296  case EVAlign::Bottom: y = r.B + descender; break;
297  }
298 
299  r = IRECT((float) x, (float) (y - ascender), (float) (x + textWidth), (float) (y + textHeight - ascender));
300 }

361 void IGraphicsCanvas::GetFontMetrics(const char* font, const char* style, double& ascenderRatio, double& EMRatio)
362 {
363  // Provides approximate font metrics for a system font (until text metrics are properly supported)
364  int size = 1000;
365  std::string fontString = GetFontString(font, style, size);
366 
367  val document = val::global("document");
368  val textSpan = document.call<val>("createElement", std::string("span"));
369  textSpan.set("innerHTML", std::string("M"));
370  textSpan["style"].set("font", fontString);
371 
372  val block = document.call<val>("createElement", std::string("div"));
373  block["style"].set("display", std::string("inline-block"));
374  block["style"].set("width", std::string("1px"));
375  block["style"].set("height", std::string("0px"));
376 
377  val div = document.call<val>("createElement", std::string("div"));
378  div.call<void>("appendChild", textSpan);
379  div.call<void>("appendChild", block);
380  document["body"].call<void>("appendChild", div);
381 
382  block["style"].set("vertical-align", std::string("baseline"));
383  double ascent = block["offsetTop"].as<double>() - textSpan["offsetTop"].as<double>();
384  double height = textSpan.call<val>("getBoundingClientRect")["height"].as<double>();
385  document["body"].call<void>("removeChild", div);
386 
387  EMRatio = size / height;
388  ascenderRatio = ascent / height;
389 }

415 bool IGraphicsCanvas::LoadAPIFont(const char* fontID, const PlatformFontPtr& font)
416 {
417  StaticStorage<Font>::Accessor storage(sFontCache);
418 
419  if (storage.Find(fontID))
420  return true;
421 
422  if (!font->IsSystem())
423  {
424  IFontDataPtr data = font->GetFontData();
425 
426  if (data->IsValid())
427  {
428  // Load the font from data
429 
430  val fontData = val(typed_memory_view(data->GetSize(), data->Get()));
431  val fontFace = val::global("FontFace").new_(std::string(fontID), fontData);
432 
433  FontDescriptor descriptor = font->GetDescriptor();
434  const double ascenderRatio = data->GetAscender() / static_cast<double>(data->GetAscender() - data->GetDescender());
435  const double EMRatio = data->GetHeightEMRatio();
436  storage.Add(new Font({descriptor->first, descriptor->second}, ascenderRatio, EMRatio), fontID);
437 
438  // Add to store and request load immediately
439 
440  fontFace.call<val>("load");
441  val::global("document")["fonts"].call<void>("add", fontFace);
442  mLoadingFonts.push_back(fontFace);
443 
444  return true;
445  }
446  }
447  else
448  {
449  FontDescriptor descriptor = font->GetDescriptor();
450  const char* fontName = descriptor->first.Get();
451  const char* styleName = descriptor->second.Get();
452 
453  if (FontExists(fontName, styleName))
454  {
455  double ascenderRatio, EMRatio;
456 
457  GetFontMetrics(descriptor->first.Get(), descriptor->second.Get(), ascenderRatio, EMRatio);
458  storage.Add(new Font({descriptor->first, descriptor->second}, ascenderRatio, EMRatio), fontID);
459  return true;
460  }
461  }
462 
463  return false;
464 }

506  int size = pBitmap->GetWidth() * pBitmap->GetHeight() * 4;

509  {

511  int height = pBitmap->GetHeight();

515  val layerCanvas = *pBitmap->GetBitmap();
516  val layerContext = layerCanvas.call<val>("getContext", std::string("2d"));
517  layerContext.call<void>("setTransform");
518 
519  if (!shadow.mDrawForeground)
520  {
521  layerContext.call<void>("clearRect", 0, 0, width, height);
522  }
523 
524  Bitmap localBitmap(width, height, pBitmap->GetScale(), pBitmap->GetDrawScale());
525  val localCanvas = *localBitmap.GetBitmap();
526  val localContext = localCanvas.call<val>("getContext", std::string("2d"));
527  val imageData = localContext.call<val>("createImageData", width, height);
528  val pixelData = imageData["data"];
529  unsigned char* in = mask.Get();
530 
531  for (auto i = 0; i < size; i++)
532  pixelData.set(i, in[i]);
533 
534  localContext.call<void>("putImageData", imageData, 0, 0);
535  IBlend blend(EBlend::SrcIn, shadow.mOpacity);
536  localContext.call<void>("rect", 0, 0, width, height);
537  localContext.call<void>("scale", scale, scale);
538  localContext.call<void>("translate", -(layer->Bounds().L + shadow.mXOffset), -(layer->Bounds().T + shadow.mYOffset));
539  SetCanvasSourcePattern(localContext, shadow.mPattern, &blend);
540  localContext.call<void>("fill");
541 
542  layerContext.set("globalCompositeOperation", "destination-over");
543  layerContext.call<void>("drawImage", localCanvas, 0, 0, width, height, x, y, width, height);
544  }
