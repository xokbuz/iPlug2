60  auto animationFunc = [&](IControl* pCaller) {
61  auto progress = pCaller->GetAnimationProgress();
62 
63  if(progress > 1.) {
64  if(mState == kExpanding)
65  {
66  mBlend.mWeight = mOpacity;
67  mState = kExpanded;
68  }
69  else if(mState == kExpanded)
70  {
71  if(GetUI()->ControlIsCaptured(mCaller))
72  {
73  mState = kExpanded;
74  SetDirty(true); // triggers animation again
75  }
76  else
77  {
78  mBlend.mWeight = mOpacity;
79  mState = kCollapsing;
80  SetDirty(true); // triggers animation again
81  }
82 
83  return; // don't remove animation
84  }
85  else if(mState == kCollapsing)
86  {
87  mBlend.mWeight = 0.;
88  Hide(true);
89  mState = kCollapsed;
90  SetDirty(false);
91  }
92 
93  pCaller->OnEndAnimation();
94  return;
95  }
96 
97  switch (mState) {
98  case kExpanding: mBlend.mWeight = (float) progress * mOpacity; break;
99  case kExpanded: mBlend.mWeight = mOpacity; break;
100  case kCollapsing: mBlend.mWeight = (float) (1.-progress) * mOpacity; break;
101  default:
102  break;
103  }
104  };
105 
106  SetActionFunction([animationFunc, this](IControl* pCaller) {
107  SetAnimation(animationFunc, 200);
108  });
109 
110  SetAnimationEndActionFunction([animationFunc, this](IControl* pCaller) {
111  if(mState == kExpanded) {
112  SetAnimation(animationFunc, 1000);
113  }
114  });
115  }

147  virtual void DrawBubble(IGraphics& g, const IRECT& r)
148  {
149  float halfCalloutSize = mCalloutSize/2.f;
150 
151  g.PathMoveTo(r.L, r.T + mRoundness);
152  g.PathArc(r.L + mRoundness, r.T + mRoundness, mRoundness, 270.f, 360.f);
153 
154  if(mArrowDir == kWest)
155  {
156  g.PathArc(r.R - mRoundness, r.T + mRoundness, mRoundness, 0.f, 90.f);
157  g.PathArc(r.R - mRoundness, r.B - mRoundness, mRoundness, 90.f, 180.f);
158  g.PathArc(r.L + mRoundness, r.B - mRoundness, mRoundness, 180.f, 270.f);
159  g.PathLineTo(r.L, r.MH() + halfCalloutSize);
160  g.PathLineTo(r.L - mCalloutSize, r.MH());
161  g.PathLineTo(r.L, r.MH() - halfCalloutSize);
162  }
163  else if(mArrowDir == kEast)
164  {
165  g.PathArc(r.R - mRoundness, r.T + mRoundness, mRoundness, 0.f, 90.f);
166  g.PathLineTo(r.R, r.MH() - halfCalloutSize);
167  g.PathLineTo(r.R + mCalloutSize, r.MH());
168  g.PathLineTo(r.R, r.MH() + halfCalloutSize);
169  g.PathArc(r.R - mRoundness, r.B - mRoundness, mRoundness, 90.f, 180.f);
170  g.PathArc(r.L + mRoundness, r.B - mRoundness, mRoundness, 180.f, 270.f);
171  }
172  else if(mArrowDir == kNorth)
173  {
174  g.PathLineTo(r.MW() - halfCalloutSize, r.T);
175  g.PathLineTo(r.MW(), r.T - mCalloutSize);
176  g.PathLineTo(r.MW() + halfCalloutSize, r.T);
177  g.PathArc(r.R - mRoundness, r.T + mRoundness, mRoundness, 0.f, 90.f);
178  g.PathArc(r.R - mRoundness, r.B - mRoundness, mRoundness, 90.f, 180.f);
179  g.PathArc(r.L + mRoundness, r.B - mRoundness, mRoundness, 180.f, 270.f);
180  }
181  else if(mArrowDir == kSouth)
182  {
183  g.PathArc(r.R - mRoundness, r.T + mRoundness, mRoundness, 0.f, 90.f);
184  g.PathArc(r.R - mRoundness, r.B - mRoundness, mRoundness, 90.f, 180.f);
185  g.PathLineTo(r.MW() + halfCalloutSize, r.B);
186  g.PathLineTo(r.MW(), r.B + mCalloutSize);
187  g.PathLineTo(r.MW() - halfCalloutSize, r.B);
188  g.PathArc(r.L + mRoundness, r.B - mRoundness, mRoundness, 180.f, 270.f);
189  }
190  g.PathClose();
191 
192  g.PathFill(mFillColor, true, &mBlend);
193  g.PathStroke(mStrokeColor, mStrokeThickness, IStrokeOptions(), &mBlend);
194  }

230  void ShowBubble(IControl* pCaller, float x, float y, const char* str, EDirection dir, IRECT minimumContentBounds, ITouchID touchID = 0)
231  {
232  if(mMaxBounds.W() == 0)
233  mMaxBounds = GetUI()->GetBounds();
234 
235  mDirection = dir;
236  mTouchId = touchID;
237 
238  mStr.Set(str);
239  IRECT contentBounds;
240  GetUI()->MeasureText(mText, str, contentBounds);
241 
242  if (!minimumContentBounds.Empty())
243  {
244  if(minimumContentBounds.W() > contentBounds.W() || minimumContentBounds.H() > contentBounds.H())
245  {
246  contentBounds = minimumContentBounds;
247  }
248  }
249 
250  contentBounds.HPad(mHorizontalPadding);
251  contentBounds.Pad(mDropShadowSize);
252  const float halfHeight = contentBounds.H() / 2.f;
253  const float halfWidth = contentBounds.W() / 2.f;
254 
255  mBubbleBounds = IRECT(x, y - halfHeight, x + contentBounds.W(), y + halfHeight);
256  IRECT controlBounds = pCaller->GetRECT();
257 
258  if(mDirection == EDirection::Horizontal)
259  {
260  const float maxR = mMaxBounds.R - mHorizontalPadding - mDropShadowSize;
261 
262  // check if it's gone off the right hand side
263  if(mBubbleBounds.R > maxR)
264  {
265  const float shiftLeft = mBubbleBounds.R-controlBounds.L;
266  mBubbleBounds.Translate(-shiftLeft, 0.f);
267  mArrowDir = EArrowDir::kEast;
268  }
269  else
270  mArrowDir = EArrowDir::kWest;
271  }
272  else
273  {
274  mBubbleBounds.Translate(-halfWidth, -halfHeight);
275 
276  // check if it's gone off the top
277  if(mBubbleBounds.T < mMaxBounds.T)
278  {
279  mArrowDir = EArrowDir::kNorth;
280  const float shiftDown = mBubbleBounds.H() + controlBounds.H();
281  mBubbleBounds.Translate(0.f, shiftDown);
282  }
283  else
284  mArrowDir = EArrowDir::kSouth;
285  }
286 
287  SetRECT(mRECT.Union(mBubbleBounds));
288 
289 // #ifndef IGRAPHICS_NANOVG
290 // if(mShadowLayer)
291 // mShadowLayer->Invalidate();
292 // #endif
293 
294  if(mState == kCollapsed)
295  {
296  Hide(false);
297  mState = kExpanding;
298  SetDirty(true);
299  }
300 
301  if(pCaller != mCaller)
302  {
303  mRECT = mBubbleBounds;
304  GetUI()->SetAllControlsDirty();
305  }
306 
307  mCaller = pCaller;
308  }
