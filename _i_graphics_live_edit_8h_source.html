128  void OnMouseDrag(float x, float y, float dX, float dY, const IMouseMod& mod) override
129  {
130  float mouseDownX, mouseDownY;
131  GetUI()->GetMouseDownPoint(mouseDownX, mouseDownY);
132 
133  if(mClickedOnControl > 0)
134  {
135  IControl* pControl = GetUI()->GetControl(mClickedOnControl);
136 
137  if(mMouseClickedOnResizeHandle)
138  {
139  IRECT r = pControl->GetRECT();
140  r.R = SnapToGrid(mMouseDownRECT.R + (x - mouseDownX));
141  r.B = SnapToGrid(mMouseDownRECT.B + (y - mouseDownY));
142 
143  if(r.R < mMouseDownRECT.L +mGridSize) r.R = mMouseDownRECT.L+mGridSize;
144  if(r.B < mMouseDownRECT.T +mGridSize) r.B = mMouseDownRECT.T+mGridSize;
145 
146  pControl->SetSize(r.W(), r.H());
147  }
148  else
149  {
150  const float x1 = SnapToGrid(mMouseDownRECT.L + (x - mouseDownX));
151  const float y1 = SnapToGrid(mMouseDownRECT.T + (y - mouseDownY));
152 
153  pControl->SetPosition(x1, y1);
154  }
155 
156  GetUI()->SetAllControlsDirty();
157  }
