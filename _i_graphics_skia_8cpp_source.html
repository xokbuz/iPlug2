161 SkPaint SkiaPaint(const IPattern& pattern, const IBlend* pBlend)
162 {
163  SkPaint paint;
164  paint.setAntiAlias(true);
165  paint.setBlendMode(SkiaBlendMode(pBlend));
166 
167  if (pattern.mType == EPatternType::Solid || pattern.NStops() < 2)
168  {
169  paint.setColor(SkiaColor(pattern.GetStop(0).mColor, pBlend));
170  }
171  else
172  {
173  double x1 = 0.0;
174  double y1 = 0.0;
175  double x2 = 0.0;
176  double y2 = 1.0;
177 
178  IMatrix m = pattern.mTransform;
179  m.Invert();
180  m.TransformPoint(x1, y1);
181  m.TransformPoint(x2, y2);
182 
183  SkPoint points[2] =
184  {
185  SkPoint::Make(x1, y1),
186  SkPoint::Make(x2, y2)
187  };
188 
189  SkColor colors[8];
190  SkScalar positions[8];
191 
192  assert(pattern.NStops() <= 8);
193 
194  for(int i = 0; i < pattern.NStops(); i++)
195  {
196  const IColorStop& stop = pattern.GetStop(i);
197  colors[i] = SkiaColor(stop.mColor, pBlend);
198  positions[i] = stop.mOffset;
199  }
200 
201  if(pattern.mType == EPatternType::Linear)
202  paint.setShader(SkGradientShader::MakeLinear(points, colors, positions, pattern.NStops(), SkiaTileMode(pattern), 0, nullptr));
203  else
204  {
205  float xd = points[0].x() - points[1].x();
206  float yd = points[0].y() - points[1].y();
207  float radius = std::sqrt(xd * xd + yd * yd);
208 
209  paint.setShader(SkGradientShader::MakeRadial(points[0], radius, colors, positions, pattern.NStops(), SkiaTileMode(pattern), 0, nullptr));
210  }
211  }
212 
213  return paint;
214 }

457 void IGraphicsSkia::PathArc(float cx, float cy, float r, float a1, float a2, EWinding winding)
458 {
459  SkPath arc;
460  float sweep = (a2 - a1);
461 
462  if (sweep >= 360.f || sweep <= -360.f)
463  {
464  arc.addCircle(cx, cy, r);
465  mMainPath.addPath(arc, mMatrix, SkPath::kAppend_AddPathMode);
466  }
467  else
468  {
469  if (winding == EWinding::CW)
470  {
471  while (sweep < 0)
472  sweep += 360.f;
473  }
474  else
475  {
476  while (sweep > 0)
477  sweep -= 360.f;
478  }
479 
480  arc.arcTo(SkRect::MakeLTRB(cx - r, cy - r, cx + r, cy + r), a1 - 90.f, sweep, false);
481  mMainPath.addPath(arc, mMatrix, SkPath::kExtend_AddPathMode);
482  }

516 void IGraphicsSkia::PrepareAndMeasureText(const IText& text, const char* str, IRECT& r, double& x, double & y, SkFont& font) const
517 {
518  SkFontMetrics metrics;
519  SkPaint paint;
520  //SkRect bounds;
521 
522  StaticStorage<Font>::Accessor storage(sFontCache);
523  Font* pFont = storage.Find(text.mFont);
524 
525  assert(pFont && "No font found - did you forget to load it?");
526 
527  font.setTypeface(pFont->mTypeface);
528  font.setHinting(SkFontHinting::kSlight);
529  font.setForceAutoHinting(false);
530  font.setSubpixel(true);
531  font.setSize(text.mSize * pFont->mData->GetHeightEMRatio());
532 
533  // Draw / measure
534  const double textWidth = font.measureText(str, strlen(str), SkTextEncoding::kUTF8, nullptr/* &bounds*/);
535  font.getMetrics(&metrics);
536 
537  const double textHeight = text.mSize;
538  const double ascender = metrics.fAscent;
539  const double descender = metrics.fDescent;
540 
541  switch (text.mAlign)
542  {
543  case EAlign::Near: x = r.L; break;
544  case EAlign::Center: x = r.MW() - (textWidth / 2.0); break;
545  case EAlign::Far: x = r.R - textWidth; break;
546  }
547 
548  switch (text.mVAlign)
549  {
550  case EVAlign::Top: y = r.T - ascender; break;
551  case EVAlign::Middle: y = r.MH() - descender + (textHeight / 2.0); break;
552  case EVAlign::Bottom: y = r.B - descender; break;
553  }
554 
555  r = IRECT((float) x, (float) y + ascender, (float) (x + textWidth), (float) (y + ascender + textHeight));
556 }

584 void IGraphicsSkia::PathStroke(const IPattern& pattern, float thickness, const IStrokeOptions& options, const IBlend* pBlend)
585 {
586  SkPaint paint = SkiaPaint(pattern, pBlend);
587  paint.setStyle(SkPaint::kStroke_Style);
588 
589  switch (options.mCapOption)
590  {
591  case ELineCap::Butt: paint.setStrokeCap(SkPaint::kButt_Cap); break;
592  case ELineCap::Round: paint.setStrokeCap(SkPaint::kRound_Cap); break;
593  case ELineCap::Square: paint.setStrokeCap(SkPaint::kSquare_Cap); break;
594  }
595 
596  switch (options.mJoinOption)
597  {
598  case ELineJoin::Miter: paint.setStrokeJoin(SkPaint::kMiter_Join); break;
599  case ELineJoin::Round: paint.setStrokeJoin(SkPaint::kRound_Join); break;
600  case ELineJoin::Bevel: paint.setStrokeJoin(SkPaint::kBevel_Join); break;
601  }
602 
603  if (options.mDash.GetCount())
604  {
605  // N.B. support odd counts by reading the array twice
606  int dashCount = options.mDash.GetCount();
607  int dashMax = dashCount & 1 ? dashCount * 2 : dashCount;
608  float dashArray[16];
609 
610  for (int i = 0; i < dashMax; i += 2)
611  {
612  dashArray[i + 0] = options.mDash.GetArray()[i % dashCount];
613  dashArray[i + 1] = options.mDash.GetArray()[(i + 1) % dashCount];
614  }
615 
616  paint.setPathEffect(SkDashPathEffect::Make(dashArray, dashMax, 0));
617  }
618 
619  paint.setStrokeWidth(thickness);
620  paint.setStrokeMiter(options.mMiterLimit);
621 
622  RenderPath(paint);
623 
624  if (!options.mPreserve)
625  mMainPath.reset();

809 void IGraphicsSkia::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
810 {
811  SkiaDrawable* pDrawable = layer->GetAPIBitmap()->GetBitmap();
812  int width = pDrawable->mSurface->width();
813  int height = pDrawable->mSurface->height();
814  size_t rowBytes = CalcRowBytes(width);

818 
819  SkMatrix m;

823  SkPixmap pixMap(info, mask.Get(), rowBytes);
824  sk_sp<SkImage> image = SkImage::MakeFromRaster(pixMap, nullptr, nullptr);
825  sk_sp<SkImage> foreground;
826 
827  // Copy the foreground if needed
828

830  foreground = pDrawable->mSurface->makeImageSnapshot();

843
