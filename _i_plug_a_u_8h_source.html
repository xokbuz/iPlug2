237  static AudioComponentMethod Lookup(SInt16 selector)
238  {
239  using Method = AudioComponentMethod;
240 
241  switch (selector)
242  {
243  case kAudioUnitInitializeSelect: return (Method)IPlugAU::AUMethodInitialize;
244  case kAudioUnitUninitializeSelect: return (Method)IPlugAU::AUMethodUninitialize;
245  case kAudioUnitGetPropertyInfoSelect: return (Method)IPlugAU::AUMethodGetPropertyInfo;
246  case kAudioUnitGetPropertySelect: return (Method)IPlugAU::AUMethodGetProperty;
247  case kAudioUnitSetPropertySelect: return (Method)IPlugAU::AUMethodSetProperty;
248 
249  case kAudioUnitAddPropertyListenerSelect: return (Method)IPlugAU::AUMethodAddPropertyListener;
250  case kAudioUnitRemovePropertyListenerSelect: return (Method)IPlugAU::AUMethodRemovePropertyListener;
251  case kAudioUnitRemovePropertyListenerWithUserDataSelect:
252  return (Method)IPlugAU::AUMethodRemovePropertyListenerWithUserData;
253 
254  case kAudioUnitAddRenderNotifySelect: return (Method)IPlugAU::AUMethodAddRenderNotify;
255  case kAudioUnitRemoveRenderNotifySelect: return (Method)IPlugAU::AUMethodRemoveRenderNotify;
256  case kAudioUnitGetParameterSelect: return (Method)IPlugAU::AUMethodGetParameter;
257  case kAudioUnitSetParameterSelect: return (Method)IPlugAU::AUMethodSetParameter;
258  case kAudioUnitScheduleParametersSelect: return (Method)IPlugAU::AUMethodScheduleParameters;
259  case kAudioUnitRenderSelect: return (Method)IPlugAU::AUMethodRender;
260  case kAudioUnitResetSelect: return (Method)IPlugAU::AUMethodReset;
261 
262  case kMusicDeviceMIDIEventSelect: return MIDIIn ? (Method)IPlugAU::AUMethodMIDIEvent : NULL;
263  case kMusicDeviceSysExSelect: return MIDIIn ? (Method)IPlugAU::AUMethodSysEx : NULL;
264 
265  default:
266  break;
267  }
268  return NULL;
269  }

271  static OSStatus Open(void* pSelf, AudioUnit compInstance)
272  {
273  AudioComponentPlugInInstance* acpi = (AudioComponentPlugInInstance *) pSelf;
274  assert(acpi);
275 
276  (*acpi->mConstruct)(&acpi->mInstanceStorage, compInstance);

294  static AudioComponentPlugInInterface* Factory(const AudioComponentDescription* pInDesc)
295  {
296  void *ptr = malloc(offsetof(AudioComponentPlugInInstance, mInstanceStorage) + sizeof(Plug));
297  AudioComponentPlugInInstance* acpi = reinterpret_cast<AudioComponentPlugInInstance*>(ptr);
298  acpi->mPlugInInterface.Open = Open;
299  acpi->mPlugInInterface.Close = Close;
300  acpi->mPlugInInterface.Lookup = Lookup;
301  acpi->mPlugInInterface.reserved = NULL;
302  acpi->mConstruct = Construct;
303  acpi->mDestruct = Destruct;
304  acpi->mPad[0] = NULL;
305  acpi->mPad[1] = NULL;
